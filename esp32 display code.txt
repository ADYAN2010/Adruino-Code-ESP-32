#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
#define SCREEN_ADDRESS 0x3C

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

#define LED_PIN 2

const char* messages[] = {
  "Coordinates",
  "Help",
  "Again Help",
  "Emergency Rescue"
};

unsigned long lastUpdate = 0;
int msgIndex = 0;
bool sirenState = false;
unsigned long lastSirenToggle = 0;
bool sirenMode = false;

#define SDA_PIN 27
#define SCL_PIN 23

void setup() {
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  Wire.begin(SDA_PIN, SCL_PIN);

  if (!display.begin(SSD1306_SWITCHCAPVCC, SCREEN_ADDRESS)) {
    while (true);
  }

  resetDisplay();
  displayMessage("System Starting...");
  delay(1500);
}

void loop() {
  // Blink blue LED every 500ms
  digitalWrite(LED_PIN, (millis() / 500) % 2);

  if (sirenMode) {
    handleSiren();
    return;
  }

  // Change message every 5 seconds (5000 ms)
  if (millis() - lastUpdate > 5000) {
    lastUpdate = millis();
    showData();
    msgIndex++;
    if (msgIndex >= sizeof(messages) / sizeof(messages[0])) {
      msgIndex = 0;
      sirenMode = true;
      lastSirenToggle = millis();
    }
  }
}

void resetDisplay() {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);
}

void displayMessage(const char* msg) {
  resetDisplay();
  display.println(msg);
  display.display();
}

void showData() {
  resetDisplay();

  float lat = 23.745 + random(-30, 30) * 0.0001;
  float lon = 90.373 + random(-30, 30) * 0.0001;

  display.print("Lat: ");
  display.println(lat, 5);

  display.print("Lon: ");
  display.println(lon, 5);

  display.print("Battery: ");
  display.println("100% ");  // fixed typo

  display.println();
  display.print("Msg:");
  display.println(messages[msgIndex]);

  display.display();
}

void handleSiren() {
  // Siren animation runs for 4 seconds
  if (millis() - lastUpdate > 4000) {
    sirenMode = false;
    resetDisplay();
    showData();
    return;
  }

  if (millis() - lastSirenToggle > 300) {
    lastSirenToggle = millis();
    sirenState = !sirenState;

    display.clearDisplay();
    display.setTextSize(2);
    display.setCursor(30, 25);

    if (sirenState) {
      display.setTextColor(SSD1306_WHITE);
      display.println("SIREN!");
    } else {
      display.setTextColor(SSD1306_BLACK);
      display.println("SIREN!");
    }
    display.display();
  }
}
